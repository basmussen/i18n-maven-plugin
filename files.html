Properties content

JSON content

  "FIRSTNAME": "Firstname",
  "LASTNAME": "Lastname"
}

<!DOCTYPE properties SYSTEM "http://java.sun.com/dtd/properties.dtd">

  <comment>Generated file customer.xml</comment>
  <entry key="LASTNAME">Lastname</entry>

  <entry key="FIRSTNAME">Firstname</entry>
</properties>
